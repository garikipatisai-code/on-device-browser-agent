# TOTP / 2FA Autofill Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Let the agent compute and enter a site's TOTP (2FA/authenticator) code itself, for a domain the user has explicitly stored a secret for in Settings — the same trust model as job-apply's stored profile autofill.

**Architecture:** Pure crypto/parsing functions in `agent/totp.ts` (no side effects, directly unit-testable), a domain-tier-gated tool `totp.get_code` that looks up a stored secret and calls those functions, a new `Settings.totpSecrets` field (persisted the same way as `domainTiers`), and one Settings UI card + one executor prompt line.

**Tech Stack:** TypeScript, Web Crypto API (`crypto.subtle` — no new dependency), Zod, Vitest.

**Spec:** `docs/superpowers/specs/2026-07-05-totp-2fa-design.md`

```ts
// TOTP (RFC 6238) / HOTP (RFC 4226) -- pure functions, no side effects, no extension APIs.
// Fixed at SHA1/6-digits/30s: the de facto standard essentially every real 2FA setup uses.
// A URI or code outside that shape is rejected with a specific reason rather than silently
// producing a wrong answer -- see parseOtpAuthUri.

export interface ParsedTotp {
  secret: string; // base32, as-is -- caller stores this; never sent to the LLM
  label: string; // issuer/account, display only, not used in the computation
}

/** Parses an otpauth://totp/... URI (the same thing a QR code encodes). */
export function parseOtpAuthUri(uri: string): { ok: true; value: ParsedTotp } | { ok: false; reason: string } {
  let url: URL;
  try {
    url = new URL(uri.trim());
  } catch {
    return { ok: false, reason: 'Not a valid URI.' };
  }
  if (url.protocol !== 'otpauth:') return { ok: false, reason: 'Not an otpauth:// URI.' };
  if (url.host !== 'totp') return { ok: false, reason: 'Only time-based (totp) codes are supported, not counter-based (hotp).' };
  const secret = url.searchParams.get('secret');
  if (!secret) return { ok: false, reason: 'URI has no secret parameter.' };
  const algorithm = url.searchParams.get('algorithm');
  if (algorithm && algorithm.toUpperCase() !== 'SHA1') {
    return { ok: false, reason: `Only SHA1 is supported (URI specifies ${algorithm}).` };
  }
  const digits = url.searchParams.get('digits');
  if (digits && digits !== '6') {
    return { ok: false, reason: `Only 6-digit codes are supported (URI specifies ${digits}).` };
  }
  const period = url.searchParams.get('period');
  if (period && period !== '30') {
    return { ok: false, reason: `Only the standard 30-second period is supported (URI specifies ${period}).` };
  }
  const label = decodeURIComponent(url.pathname.replace(/^\/+/, '')) || url.searchParams.get('issuer') || 'Unnamed';
  return { ok: true, value: { secret, label } };
}

const BASE32_ALPHABET = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ234567';

/** RFC 4648 base32 decode. Case-insensitive, ignores '=' padding and whitespace (how TOTP
 *  secrets are commonly copy-pasted). Throws on a character outside the alphabet. */
export function base32Decode(input: string): Uint8Array {
  const clean = input.toUpperCase().replace(/[=\s]/g, '');
  const bytes: number[] = [];
  let bits = 0;
  let value = 0;
  for (const ch of clean) {
    const idx = BASE32_ALPHABET.indexOf(ch);
    if (idx === -1) throw new Error(`Invalid base32 character: ${ch}`);
    value = (value << 5) | idx;
    bits += 5;
    if (bits >= 8) {
      bits -= 8;
      bytes.push((value >> bits) & 0xff);
    }
  }
  return new Uint8Array(bytes);
}

/** RFC 6238 TOTP over RFC 4226 HOTP. `unixSeconds` is injectable for deterministic testing;
 *  defaults to real time for actual use. */
export async function computeTotpCode(
  secretBytes: Uint8Array,
  unixSeconds: number = Math.floor(Date.now() / 1000),
): Promise<string> {
  const counter = Math.floor(unixSeconds / 30);
  const counterBytes = new Uint8Array(8);
  let c = counter;
  for (let i = 7; i >= 0; i--) {
    counterBytes[i] = c & 0xff;
    c = Math.floor(c / 256);
  }
  const key = await crypto.subtle.importKey('raw', secretBytes, { name: 'HMAC', hash: 'SHA-1' }, false, ['sign']);
  const sig = new Uint8Array(await crypto.subtle.sign('HMAC', key, counterBytes));
  const offset = sig[19] & 0xf;
  const code =
    ((sig[offset] & 0x7f) << 24) |
    ((sig[offset + 1] & 0xff) << 16) |
    ((sig[offset + 2] & 0xff) << 8) |
    (sig[offset + 3] & 0xff);
  return String(code % 1_000_000).padStart(6, '0');
}
```

```ts
// totp.get_code: computes the current 2FA code for a domain the user has stored a TOTP secret
// for (Settings -> Two-factor codes). No CDP/chrome.debugger involved -- this is a Settings
// lookup + local crypto, not a browser action -- domain-tier gated anyway, same as every
// mutating tool, since entering a 2FA code is a consequential action on the page.
import { z } from 'zod';
import type { ToolDefDescriptor } from '../registry';
import { assertCanAct, hostFor } from '@/agent/safety/domain_tiers';
import { base32Decode, computeTotpCode } from '@/agent/totp';
import { tabUrl } from './actions';

export const totpGetCodeTool: ToolDefDescriptor<{ tabId: number }> = {
  name: 'totp.get_code',
  description:
    'Get the current 6-digit two-factor/verification code for the site open in this tab, if a TOTP secret is stored for its domain (Settings → Two-factor codes). Returns just the code — type it with tab.type immediately after, since it expires in about 30 seconds. Fails with a clear message if no secret is stored for this domain.',
  argsSchema: z.object({ tabId: z.number().int() }),
  async dispatch({ tabId }, ctx) {
    const url = await tabUrl(tabId);
    assertCanAct(url, 'click-only', ctx.settings.domainTiers, ctx.settings.bypassDomainTiers);
    const host = hostFor(url);
    const entry = ctx.settings.totpSecrets?.[host];
    if (!entry) {
      return {
        ok: false,
        content: `No TOTP secret stored for ${host}. Add one in Settings → Two-factor codes, or ask the user for the code.`,
      };
    }
    const code = await computeTotpCode(base32Decode(entry.secret));
    return { ok: true, content: code };
  },
};
```
